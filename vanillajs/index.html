# "The largest value smaller or equal to target"

def binarySearchLeft(arr):
    l, r = 0, len(arr) - 1
    res = -1
    while l <= r:
        m = l + (r - l) // 2
        if arr[m] <= target:
            l = m + 1
            res = m
        else:
            r = m - 1
    return res
